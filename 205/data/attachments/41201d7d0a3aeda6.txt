var jsonData = pm.response.json();

pm.test("Response status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response indicates failure", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Error message about the required specialization is returned", function () {
    pm.expect(jsonData.error).to.eql("AOS is required");
});
